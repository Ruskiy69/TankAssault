108 void CL_MenuManager::MainMenu()
109 {
110  mp_ActiveMenu = &m_MainMenu;
111 
112  int status = m_MainMenu.Update();
113 
114  if(status == -1)
115  return;
116  else if(status == 0)
117  m_state = game::e_INTRO;
118  else if(status == 1)
119  m_state = game::e_OPTIONSMENU;
120  else if(status == 2)
121  m_state = game::e_QUIT;
122 
123  // I pulled a tiny trick here. Technically, if the game state changes,
124  // there should be a delay of 200ms so that the mouse click isn't
125  // registered again (user can lift finger off of mouse button).
126  // So, if there was no state change, return from the function, but if
127  // there was, do the delay.
128  SDL_Delay(200);
129 }
