98 void Game_MenuManager::MainMenu()
99 {
100  mp_ActiveMenu = &m_MainMenu;
101 
102  int status = m_MainMenu.Update();
103 
104  if(status == -1)
105  return;
106  else if(status == 0)
107  m_state = game::e_INTRO;
108  else if(status == 1)
109  m_state = game::e_OPTIONSMENU;
110  else if(status == 2)
111  m_state = game::e_QUIT;
112 
113  // I pulled a tiny trick here. Technically, if the game state changes,
114  // there should be a delay of 200ms so that the mouse click isn't
115  // registered again (user can lift finger off of mouse button).
116  // So, if there was no state change, return from the function, but if
117  // there was, do the delay.
118  SDL_Delay(200);
119 }
