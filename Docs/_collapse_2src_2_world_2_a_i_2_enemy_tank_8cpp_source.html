138 void CEnemyTank::OnPatrolling()
139 {
140  // Whelp, this shouldn't be happening...
141  if(m_tactic != e_PATROLLING)
142  return;
143 
144  // If we're done with our path, just go back to where we started from.
145  if(m_state & e_DONE)
146  {
147  this->RemoveState(e_DONE);
148  obj::CEntity* p_Dest = mp_Level->GetObjectiveMap().GetNearestPOI(
149  m_Tank.GetPosition());
150  this->SetDestination(p_Dest->GetPosition());
151  if(mp_DestinationTile == NULL)
152  this->AddState(e_DONE);
153  else
154  this->AddState(e_PATHFINDING);
155  }
156 
157  // Scan the area for hostiles.
158  // This is done by repeatedly rotating the tank tower 90 degrees in
159  // both directions from the tank front-center position, with a slight
160  // pause upon reaching the end-points.
161 
162  // This is the direction in which to rotate the tower.
163  static float rotation_speed = 3.0f;
164 
165  // Check that the rotation angle is within the closed interval
166  // [Tower Rotation - 90, Tower Rotation + 90]
167  if((m_Tower.GetRotationAngle() >= 90.0f + m_Tank.GetRotationAngle() &&
168  rotation_speed == 3.0f) ||
169 
170  (m_Tower.GetRotationAngle() <= m_Tank.GetRotationAngle() - 90.0f &&
171  rotation_speed == -3.0f))
172  {
173  rotation_speed = -rotation_speed;
174  }
175  else
176  {
177  this->RotateTower(rotation_speed);
178  }
179 
180  // Check if player is in line-of-sight. If he is, calculate the angle
181  // toward him (for the pathfinding), change the tactic to
182  // the hostile e_ATTACKING tactic, reset the state, add pathfinding,
183  // and set the new path as the player location.
184  if(m_LineOfSight.CheckCollision(m_Player.GetCollisionBox()))
185  {
186 #ifdef _DEBUG
187  printf("[DEBUG] Player spotted!\n");
188 #endif // _DEBUG
189 
190  // Calculate axis toward player
191  m_axis_to_player = math::deg(
192  atan2(m_Tank.GetX() - m_Player.GetPosition().x,
193  m_Tank.GetY() - m_Player.GetPosition().y));
194 
195  // Attack the newly found enemy!
196  this->SetTactic(e_ATTACKING);
197 
198  // Find a path to the enemy
199  this->SetDestination(m_Player.GetPosition());

257  if(m_PlayerLastSeen != m_Player.GetPosition())

260  }

384 void CEnemyTank::FollowPath()
385 {
386  if(!(m_state & e_PATHFINDING))
387  return;
388 
389  else if(m_state & e_NO_PATH)
390  {
392  this->SetDestination(mp_Level->GetObjectiveMap().
393  GetNearestPOI(this->GetPosition())->GetPosition());
394  this->ResetState();
395  this->SetState(e_PATHFINDING);
396  printf("[DEBUG] No path found!\n");
397  return;
398  }
399 
400  else if(m_state & e_DONE)
401  {
402  printf("[DEBUG] Path following complete.\n");
403  return;
404  }
405 
406  else if(mp_DestinationTile == NULL)
407  {
408  printf("[DEBUG] mp_DestinationTile is NULL!\n");
409  return;
410  }
411 
412  // If we have reached our current tile destination.
413  else if(mp_DestinationTile->CheckCollision(
414  math::CRect(m_Tank.GetX() + 16, m_Tank.GetY(), 32, 8)))
415  {
416 #ifdef _DEBUG
417  printf("[DEBUG] Next tile called due to collision.\n");
418 #endif // _DEBUG
419 
420  // Find the next one
421  mp_DestinationTile = m_Pathfinder.NextTile();
422 
423  if(mp_DestinationTile == NULL)
424  {
425  this->RemoveState(e_PATHFINDING);
426  this->AddState(e_DONE);
427  return;
428  }
429  else
430  {
431  // Calculate angle toward destination
432  m_axis_to_path = math::deg(
433  atan2(m_Tank.GetY() - mp_DestinationTile->GetY(),
434  m_Tank.GetX() - mp_DestinationTile->GetX()));
435  m_axis_to_path += 90.0f;
436 
437  // Keep the angle in the interval [0, 360]
438  if(m_axis_to_path < 0.0f)
439  m_axis_to_path = 180.0f + 180.0f + m_axis_to_path;
440  while(m_axis_to_path >= 360.0f)
441  m_axis_to_path -= 360.0f;
442 
443 #ifdef _DEBUG
444  printf("[DEBUG] Axis toward next tile: %0.1f\n", m_axis_to_path);
445 #endif // _DEBUG
446  }
447  }
448 
449  // Move toward the destination
450  else
451  {
452  // Our angle
453  float current_angle = m_Tank.GetRotationAngle();
454 
455  // Adjust so it's in a 3 degree range
456  while((int)m_axis_to_path % 3 != 0) m_axis_to_path--;
457 
458  // If we're not axis-aligned
459  if(!(current_angle + 1.0f >= m_axis_to_path &&
460  current_angle - 1.0f <= m_axis_to_path))
461  {
462  if(current_angle < m_axis_to_path)
463  this->Turn(3.0f);
464  else
465  this->Turn(-3.0f);
466  }
467  else
468  {
469  this->Drive(-3.0f);
470  if(mp_Level->GetCollisionMap().FindTile(this->GetPosition()) != NULL)
471  this->Drive(3.0f);
472  }
473  }
474 }

481 void CEnemyTank::UpdateLOS()
482 {
483  // Refreshes barrel pos.
484  this->RotateTower(0);
485 
486  // Locate tank center
487  math::CVector2 Tank_Center(this->GetPosition().x + 32, this->GetPosition().y + 32);
488 
489  // Reset LOS to center-top, 8 tiles out.
490  m_LineOfSight.Start = this->GetBarrelPosition();
491  m_LineOfSight.End.Move(this->GetPosition().x - 2 + 32, this->GetPosition().y + 256);
492 
493  // Rotate around center
494  m_LineOfSight = m_LineOfSight - Tank_Center;
495  m_LineOfSight.End.Rotate(math::rad(m_Tower.GetRotationAngle()));
496  m_LineOfSight = m_LineOfSight + Tank_Center;
497 
498  // Cut off at walls
499  const std::vector<obj::CGameObject*>& p_allTiles =
500  mp_Level->GetCollisionMap().GetTiles();
501  math::CVector2 Intersection;
502  for(size_t i = 0; i < p_allTiles.size(); ++i)
503  {
504  if(m_LineOfSight.CheckCollision(p_allTiles[i]->GetCollisionBox(), &Intersection))
505  m_LineOfSight.End = Intersection;
506  }
507 }
