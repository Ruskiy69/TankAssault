141 void AI_Tank::OnPatrolling()
142 {
143  // Whelp, this shouldn't be happening...
144  if(m_tactic != e_PATROLLING)

148  if(m_state & e_DONE)

159  // Scan the area for hostiles.
160  // This is done by repeatedly rotating the tank tower 90 degrees in

164  // This is the direction in which to rotate the tower.

169  if((mp_Tower->GetRotation() >= 90.0f + mp_Tank->GetRotation() &&

184  // the hostile e_ATTACKING tactic, reset the state, add pathfinding,
185  // and set the new path as the player location.

342 void AI_Tank::FollowPath()
343 {
344  if(!(m_state & e_PATHFINDING))
345  return;
346 
347  else if(m_state & e_NO_PATH)
348  {
350  //this->SetDestination(m_ObjectiveMap.FindNearestPOI(this->GetPosition()));
351  //this->ResetState();
352  //this->SetState(e_PATHFINDING);
353  printf("[DEBUG] No path found!\n");
354  return;
355  }
356 
357  else if(m_state & e_DONE)
358  {
359  printf("[DEBUG] Path following complete.\n");
360  return;
361  }
362 
363  else if(mp_DestinationTile == NULL)
364  {
365  printf("[DEBUG] mp_DestinationTile is NULL!\n");
366  return;
367  }
368 
369  // If we have reached our current tile destination.
370  else if(mp_DestinationTile->CheckCollision(
371  math::ML_Rect(mp_Tank->GetX() + 16, mp_Tank->GetY(), 32, 8)))
372  {
373 #ifdef _DEBUG
374  printf("[DEBUG] Next tile called due to collision.\n");
375 #endif // _DEBUG
376 
377  // Find the next one
378  mp_DestinationTile = m_Pathfinder.NextTile();
379 
380  if(mp_DestinationTile == NULL)
381  {
382  this->RemoveState(e_PATHFINDING);
383  this->AddState(e_DONE);
384  return;
385  }
386  else
387  {
388  // Calculate angle toward destination
389  m_axis_to_path = math::deg(
390  atan2(mp_Tank->GetX() - mp_DestinationTile->GetX(),
391  mp_Tank->GetY() - mp_DestinationTile->GetY()));
392 
393  // Keep the angle in the interval [0, 360]
394  if(m_axis_to_path < 0.0f)
395  m_axis_to_path = 180.0f + 180.0f + m_axis_to_path;
396  while(m_axis_to_path >= 360.0f)
397  m_axis_to_path -= 360.0f;
398 
399 #ifdef _DEBUG
400  printf("[DEBUG] Axis toward next tile: %0.1f\n", m_axis_to_path);
401 #endif // _DEBUG
402  }
403  }
404 
405  // Move toward the destination
406  else
407  {
408  // Our angle
409  float current_angle = mp_Tank->GetRotation() + 180.0f;
410 
411  // Adjust so it's in a 3 degree range
412  while((int)m_axis_to_path % 3 != 0) m_axis_to_path--;
413 
414  // If we're not axis-aligned
415  if(!(current_angle + 1.0f >= m_axis_to_path &&
416  current_angle - 1.0f <= m_axis_to_path))
417  {
418  if(current_angle < m_axis_to_path)
419  this->Turn(3.0f);
420  else
421  this->Turn(-3.0f);
422  }
423  else
424  {
425  this->Drive(3.0f);
426  if(m_CollisionMap.FindTile(this->GetPosition()) != NULL)
427  this->Drive(-3.0f);
428  }
429  }
