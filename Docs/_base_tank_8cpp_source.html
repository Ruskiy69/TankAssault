102 void Obj_Tank::Aim(const float x, const float y)
103 {
104  // Triangles eh? Take the arctangent to determine the shot angle (in degrees)
105  // The plus 180 at the end is necessary, though I forgot to add a comment explaining why...
106  // I think it's due to the fact that the tower is stored facing down in the .png image.
107  float angle = math::deg(atan2(mp_Tower->GetX() - x, mp_Tower->GetY() - y)) + 180;
108 
109  // A tiny optimization, I guess, so that the new barrel isn't calculated.
110  // Yes, I'm aware of floating point inaccuracies.
111  if(mp_Tower->GetRotation() == angle)
112  return;
113 
114  mp_Tower->Rotate(angle);
115 
125  // Convert to radians for sin and cos in math::ML_Vector2::Rotate() to function.
126  angle = math::rad(angle - 180.0f);
127 
128  // Find the center of the tank, this is the origin.
129  float center_x = mp_Tank->GetX() + (mp_Tank->GetW() / 2);
130  float center_y = mp_Tank->GetY() + (mp_Tank->GetH() / 2);
131 
132  // Reset the barrel position to the top point.
133  m_BarrelPosition.Move(mp_Tank->GetX() - 2 + mp_Tank->GetW() / 2, mp_Tank->GetY());
134 
135  // Subtract the center from the barrel.
136  math::ML_Vector2 Center(center_x, center_y);
137  math::ML_Vector2 Final = m_BarrelPosition - Center;
138 
139  // Rotate
140  Final.Rotate(angle);
141 
142  // Add the center back to the barrel position.
143  Final = Final + Center;
144 
145  m_BarrelPosition.Move(Final);
146 }
