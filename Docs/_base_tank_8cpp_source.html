111 void Obj_Tank::Aim(const float x, const float y)
112 {
113  // Triangles eh? Take the arctangent to determine the shot angle (in degrees)
114  // The plus 180 at the end is necessary, though I forgot to add a comment explaining why...
115  // I think it's due to the fact that the tower is stored facing down in the .png image.
116  float angle = math::ToDegrees(atan2(m_Tower.GetX() - x, m_Tower.GetY() - y)) + 180;
117 
118  // A tiny optimization, I guess, so that the new barrel isn't calculated.
119  // Yes, I'm aware of floating point inaccuracies.
120  if(m_Tower.GetRotation() == angle)
121  return;
122 
123  m_Tower.Rotate(angle);
124 
134  // Convert to radians for sin and cos in math::Vector2::Rotate() to function.
135  angle = math::ToRadians(angle - 180.0f);
136 
137  // Find the center of the tank, this is the origin.
138  float center_x = m_Tank.GetX() + (m_Tank.GetW() / 2);
139  float center_y = m_Tank.GetY() + (m_Tank.GetH() / 2);
140 
141  // Reset the barrel position to the top point.
142  m_BarrelPosition.Move(m_Tank.GetX() - 2 + m_Tank.GetW() / 2, m_Tank.GetY());
143 
144  // Subtract the center from the barrel.
145  math::Vector2 Center(center_x, center_y);
146  math::Vector2 Final = m_BarrelPosition - Center;
147 
148  // Rotate
149  Final.Rotate(angle);
150 
151  // Add the center back to the barrel position.
152  Final = Final + Center;
153 
154  m_BarrelPosition.Move(Final);
155 }
