106 void Obj_Tank::Aim(const float x, const float y)
107 {
108  // Triangles eh? Take the arctangent to determine the shot angle (in degrees)
109  // The plus 180 at the end is necessary, though I forgot to add a comment explaining why...
110  // I think it's due to the fact that the tower is stored facing down in the .png image.
111  float angle = math::deg(atan2(mp_Tower->GetX() - x, mp_Tower->GetY() - y)) + 180;
112 
113  // A tiny optimization, I guess, so that the new barrel isn't calculated.
114  // Yes, I'm aware of floating point inaccuracies.
115  if(mp_Tower->GetRotation() == angle)
116  return;
117 
118  mp_Tower->Rotate(angle);
119 
129  // Convert to radians for sin and cos in math::ML_Vector2::Rotate() to function.
130  angle = math::rad(angle - 180.0f);
131 
132  // Find the center of the tank, this is the origin.
133  float center_x = mp_Tank->GetX() + (mp_Tank->GetW() / 2);
134  float center_y = mp_Tank->GetY() + (mp_Tank->GetH() / 2);
135 
136  // Reset the barrel position to the top point.
137  m_BarrelPosition.Move(mp_Tank->GetX() - 2 + mp_Tank->GetW() / 2, mp_Tank->GetY());
138 
139  // Subtract the center from the barrel.
140  math::ML_Vector2 Center(center_x, center_y);
141  math::ML_Vector2 Final = m_BarrelPosition - Center;
142 
143  // Rotate
144  Final.Rotate(angle);
145 
146  // Add the center back to the barrel position.
147  Final = Final + Center;
148 
149  m_BarrelPosition.Move(Final);
150 }
