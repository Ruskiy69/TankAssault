101 void CTank::RotateTower(const float deg)
102 {
103  // Rotate the tower entity
104  m_Tower.Rotate_Rate(deg);
105 
106  // Convert to radians for sin and cos in math::CVector2::Rotate() to function.
107  float rad = -math::rad(m_Tower.GetRotationAngle());
108 
118  // Find the center of the tank, this is the origin.
119  float center_x = m_Tank.GetX() + (m_Tank.GetW() / 2);
120  float center_y = m_Tank.GetY() + (m_Tank.GetH() / 2);
121 
122  // Reset the barrel position to the top point.
123  m_BarrelPosition.Move(center_x - 2, center_y + 32);
124 
125  // Subtract the center from the barrel.
126  math::CVector2 Center(center_x, center_y);
127  math::CVector2 Final = m_BarrelPosition - Center;
128 
129  // Rotate
130  Final.Rotate(rad);
131 
132  // Add the center back to the barrel position.
133  Final = Final + Center;
134  m_BarrelPosition = Final;
135 }
