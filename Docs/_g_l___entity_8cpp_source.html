453 void GL_Entity::Render() const
454 {
455  // Alpha channel is gucci
456  GLboolean blend = glIsEnabled(GL_BLEND);
457 
458  glEnable(GL_BLEND);
459  glDisable(GL_DEPTH_TEST);
460  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
461 
462  // Render the primitive with the texture on top.
463  // First, rotate around the origin.
464  glMatrixMode(GL_MODELVIEW);
465  glPushMatrix();
466  glLoadIdentity();
467 
468  glTranslatef(this->GetX() + this->GetW() / 2,
469  this->GetY() + this->GetH() / 2, 0.0f);
470  glRotatef(-m_rotation, 0.0f, 0.0f, 1.0f);
471  glTranslatef(-this->GetX() - this->GetW() / 2,
472  -this->GetY() - this->GetH() / 2, 0.0f);
473 
474  glBindTexture(GL_TEXTURE_2D, m_texture);
475 
476  // Draw the primitive
477  glBegin(GL_QUADS);
478  // Top left corner
479  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y);
480  glVertex2f(m_Position.x, m_Position.y);
481 
482  // Top right corner
483  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
484  m_RenderDimensions.y);
485  glVertex2f(m_Position.x + m_CollisionBox.w, m_Position.y);
486 
487  // Bottom right corner
488  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
489  m_RenderDimensions.y + m_RenderDimensions.h);

491 
492  // Bottom left corner
493  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y + m_RenderDimensions.h);
494  glVertex2f(m_Position.x, m_Position.y + m_CollisionBox.h);
495  glEnd();
