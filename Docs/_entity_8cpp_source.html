433 void GL_Entity::Render() const
434 {
435  // Alpha channel is Gucci
436  GLboolean blend = glIsEnabled(GL_BLEND);
437 
438  glEnable(GL_BLEND);
439  glDisable(GL_DEPTH_TEST);
440  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
441 
442  // Render the primitive with the texture on top.
443  // First, rotate around the origin.
444  glMatrixMode(GL_MODELVIEW);
445  glPushMatrix();
446  glLoadIdentity();
447 
448  glTranslatef(this->GetX() + this->GetW() / 2,
449  this->GetY() + this->GetH() / 2, 0.0f);
450  glRotatef(-m_rotation, 0.0f, 0.0f, 1.0f);
451  glTranslatef(-this->GetX() - this->GetW() / 2,
452  -this->GetY() - this->GetH() / 2, 0.0f);
453 
454  glActiveTexture(GL_TEXTURE0);
455  glBindTexture(GL_TEXTURE_2D, m_texture);
456 
457  // Draw the primitive
458  glBegin(GL_QUADS);
459  // Top left corner
460  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y);
461  glVertex2f(m_Position.x, m_Position.y);
462 
463  // Top right corner
464  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
465  m_RenderDimensions.y);
466  glVertex2f(m_Position.x + m_CollisionBox.w, m_Position.y);
467 
468  // Bottom right corner
469  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
470  m_RenderDimensions.y + m_RenderDimensions.h);
471  glVertex2f(m_Position.x + m_CollisionBox.w, m_Position.y + m_CollisionBox.h);
472 
473  // Bottom left corner
474  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y + m_RenderDimensions.h);
475  glVertex2f(m_Position.x, m_Position.y + m_CollisionBox.h);
476  glEnd();
477 
478  // Disable blending if it was disabled originally
479  if(!blend)
480  {
481  glDisable(GL_BLEND);
482  glEnable(GL_DEPTH_TEST);
483  }
484 
485  glPopMatrix();
486  glBindTexture(GL_TEXTURE_2D, 0);
487 }
