430 void GL_Entity::Render() const
431 {
432  // Alpha channel is gucci
433  GLboolean blend = glIsEnabled(GL_BLEND);
434 
435  glEnable(GL_BLEND);
436  glDisable(GL_DEPTH_TEST);
437  glBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
438 
439  // Render the primitive with the texture on top.
440  // First, rotate around the origin.
441  glMatrixMode(GL_MODELVIEW);
442  glPushMatrix();
443  glLoadIdentity();
444 
445  glTranslatef(this->GetX() + this->GetW() / 2,
446  this->GetY() + this->GetH() / 2, 0.0f);
447  glRotatef(-m_rotation, 0.0f, 0.0f, 1.0f);
448  glTranslatef(-this->GetX() - this->GetW() / 2,
449  -this->GetY() - this->GetH() / 2, 0.0f);
450 
451  glBindTexture(GL_TEXTURE_2D, m_texture);
452 
453  // Draw the primitive
454  glBegin(GL_QUADS);
455  // Top left corner
456  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y);
457  glVertex2f(m_Position.x, m_Position.y);
458 
459  // Top right corner
460  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
461  m_RenderDimensions.y);
462  glVertex2f(m_Position.x + m_CollisionBox.w, m_Position.y);
463 
464  // Bottom right corner
465  glTexCoord2f(m_RenderDimensions.x + m_RenderDimensions.w,
466  m_RenderDimensions.y + m_RenderDimensions.h);
467  glVertex2f(m_Position.x + m_CollisionBox.w, m_Position.y + m_CollisionBox.h);
468 
469  // Bottom left corner
470  glTexCoord2f(m_RenderDimensions.x, m_RenderDimensions.y + m_RenderDimensions.h);
471  glVertex2f(m_Position.x, m_Position.y + m_CollisionBox.h);
472  glEnd();
473 
474  // Disable blending if it was disabled originally
475  if(!blend)

479  }
