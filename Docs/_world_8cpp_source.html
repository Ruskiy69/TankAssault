405 bool CL_World::SpawnEnemy()

409  for(ai::AI_Enemies::iterator i = ai::AI_Enemy::p_allEnemies.begin();
410  i != ai::AI_Enemy::p_allEnemies.end(); ++i)
411  {
412  p_allObjs.push_back((asset::GL_Entity*)(*i)->GetMainEntity());
413  }
414 
415  asset::GL_Entity* p_Spawn = m_ObjectiveMap.GetAvailableEnemySpawn(p_allObjs);
416  if(p_Spawn == NULL)
417  return false;
418 
419  asset::GL_Entity* p_Dest = m_ObjectiveMap.GetNearestPOI(p_Spawn->GetPosition());
420  if(p_Dest == NULL)
421  return false;
422 
423  ai::AI_Enemy* p_Enemy = new ai::AI_Tank(m_TerrainMap,
424  m_CollisionMap, m_ObjectiveMap, m_Player);
425 
426  p_Enemy->Spawn(p_Spawn->GetPosition());
427  p_Enemy->SetDestination(p_Dest->GetPosition() +
428  math::ML_Vector2(1.0f, 1.0f));
429 
430  obj::Obj_pBullets p_enemyBullets;
431  p_enemyBullets.clear();
432  p_allObjs.clear();
433  mp_enemyBullets.push_back(p_enemyBullets);
434 
435  return true;
